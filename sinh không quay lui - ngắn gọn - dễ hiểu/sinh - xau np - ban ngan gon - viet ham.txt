#include <bits/stdc++.h>
using namespace std;

int B[25], n;

void in() {
    for (int i = 0; i < n; i++) cout << B[i] << " ";
    cout << endl;
}

void sinh() {
    fill(B, B + n, 0); // khởi tạo xâu đầu tiên

    while(true) {
        in();

        // tăng nhị phân
        int i = n - 1;
        while(i >= 0 && B[i] == 1) {
            B[i] = 0;
            i--;
        }
        if(i < 0) break;
        B[i] = 1;
    }
}

int main() {
    cin >> n;
    sinh();
}
