#include <bits/stdc++.h>
using namespace std;

int B[25], n;

// Hàm in ra xâu hiện tại
void in() {
    for (int i = 0; i < n; i++) cout << B[i] << " ";
    cout << endl;
}

// Hàm sinh tất cả xâu nhị phân không đệ quy
void sinh() {
    while (true) {
        // In ra xâu hiện tại
        in();

        // Tăng nhị phân
        int i = n - 1;
        while (i >= 0 && B[i] == 1) {
            B[i] = 0;
            i--;
        }
        if (i < 0) break; // đã sinh hết
        B[i] = 1;
    }
}

int main() {
    cin >> n;
    fill(B, B + n, 0); // khởi tạo mảng
    sinh();             // gọi hàm sinh
}
