#include<bits/stdc++.h>
using namespace std;

int n, A[15], ok;

void ktao(){
    // Khởi tạo hoán vị đầu tiên: 1 2 ... n
    for(int i=1; i<=n; i++) A[i] = i;
}

void in(){
    for(int i=1; i<=n; i++) cout << A[i] << " ";
    cout << endl;
}

void sinh(){
    int i = n - 1;
    // Tìm i sao cho A[i] < A[i+1]
    while(i >= 1 && A[i] > A[i+1]) i--;
    
    if(i == 0){
        ok = 0; // Cấu hình cuối cùng: n n-1 ... 1
    } else {
        // Tìm j > i sao cho A[j] > A[i] và nhỏ nhất có thể
        int j = n;
        while(A[j] < A[i]) j--;
        swap(A[i], A[j]);
        // Lật ngược đoạn từ i+1 đến n
        int l = i + 1, r = n;
        while(l < r){
            swap(A[l], A[r]);
            l++; r--;
        }
    }
}

int main(){
    cin >> n;
    ok = 1;
    ktao();
    while(ok){
        in();
        sinh();
    }
    return 0;
}
